def binary_iterative(data, target):
    left, right, steps = 0, len(data)-1, 0
    while left <= right:
        steps += 1
        mid = (left + right) // 2
        if data[mid] == target:
            return mid, steps
        if data[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1, steps

from collections import deque

class LunchQueue:
    def __init__(self):
        self.q = deque()

    def enter(self, student):
        self.q.append(student)

    def serve(self):
        return self.q.popleft() if self.q else None

line = LunchQueue()
for s in ["민수", "서연", "지훈"]:
    line.enter(s)
print(line.serve())

def balanced(text):
    pairs = {")": "(", "]": "[", "}": "{"}
    stack = []
    for ch in text:
        if ch in "([{":
            stack.append(ch)
        elif ch in pairs:
            if not stack or stack.pop() != pairs[ch]:
                return False
    return not stack

print(balanced("{a + [b * (c+d)]}"))

    for index, ch in enumerate(expression):
        if ch == "(":
            stack.append(index)

def spread(graph, initial, steps):
    infected = set(initial)
    frontier = set(initial)
    for _ in range(steps):
        new = {n for x in frontier for n in graph[x]} - infected
        infected |= new
        frontier = new
    return infected

    left, right, count = 0, len(data)-1, 0
    while left <= right:
        count += 1
        mid = (left+right)//2
        if data[mid] == target:
            return mid, count
        if data[mid] < target: left = mid+1
        else: right = mid-1
    return -1, count

    left, right = 0, len(data)-1
    while left <= right:
        mid = (left+right)//2

        if data[mid] < target: left = mid+1
        else: right = mid-1
    return -1
