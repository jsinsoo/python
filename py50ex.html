Python 프로젝트 예시 코드
import hashlib

def mask_name(name):
    return name[0] + "*" * (len(name) - 1)

def mask_phone(phone):
    parts = phone.split("-")
    return f"{parts[0]}-****-{parts[2]}" if len(parts) == 3 else "형식 오류"

def pseudonymize(text, salt="school"):
    return hashlib.sha256((salt + text).encode()).hexdigest()[:12]

licenses = {
    "학교사진.jpg": "CC BY-NC",
    "아이콘.png": "CC0",
    "코드.py": "MIT"
}

print(mask_name("홍길동"))
print(mask_phone("010-1234-5678"))
print(pseudonymize("1101_홍길동"))
for file, license_name in licenses.items():
    print(file, "→", license_name)

checks = {
    "출처를 기록했는가": True,
    "개인정보를 제거했는가": True,
    "편향 가능성을 검토했는가": False,

    "저작권 침해 가능성을 확인했는가": False

if score < len(checks):
    print("보완이 필요한 항목을 수정한 뒤 결과물을 사용하세요.")

12정02-03 · 인구·기후 데이터 분석 시각화

지역별 인구 증감률과 기온 변화를 분석하고 그래프로 표현한다.

A형 과세특 연결

데이터의 추세와 상관관계를 분석하고 시각화를 통해 시사점을 제안함.

B형 과세특 연결

Matplotlib로 적절한 그래프를 구현하여 데이터 특징을 전달함.

Python 프로젝트 예시 코드
# 사전 설치: pip install pandas matplotlib
import pandas as pd
import matplotlib.pyplot as plt

df = pd.DataFrame({
    "지역": ["A", "B", "C", "D"],
    "2020인구": [120000, 95000, 80000, 70000],
    "2025인구": [116000, 90000, 76000, 65000],
    "평균기온": [13.8, 14.1, 14.5, 14.8]
})
df["증감률"] = (df["2025인구"] - df["2020인구"]) / df["2020인구"] * 100
print(df.round(2))
print("상관계수:", df["증감률"].corr(df["평균기온"]))
plt.scatter(df["평균기온"], df["증감률"])
for _, row in df.iterrows():
    plt.text(row["평균기온"], row["증감률"], row["지역"])
plt.xlabel("평균기온")
plt.ylabel("인구 증감률(%)")
plt.title("기후와 인구 변화 관계 탐색")

from time import perf_counter

def bubble_sort(data):
    a = data[:]
    for end in range(len(a)-1, 0, -1):
        for i in range(end):
            if a[i] > a[i+1]:
                a[i], a[i+1] = a[i+1], a[i]
    return a

data = [random.randint(0, 10000) for _ in range(1000)]
for fn in [bubble_sort, sorted]:
    start = perf_counter()
    fn(data)
    print(fn.__name__, perf_counter() - start)

B형 과세특 연결

Python 프로젝트 예시 코드
from collections import defaultdict

records = [
    {"group": "A", "actual": 1, "pred": 1},
    {"group": "A", "actual": 1, "pred": 0},
    {"group": "A", "actual": 0, "pred": 0},
    {"group": "B", "actual": 1, "pred": 1},
    {"group": "B", "actual": 0, "pred": 1},
    {"group": "B", "actual": 0, "pred": 0}
]
stats = defaultdict(lambda: {"correct": 0, "total": 0})
for r in records:
    stats[r["group"]]["total"] += 1
    stats[r["group"]]["correct"] += int(r["actual"] == r["pred"])
accuracies = {}
for group, s in stats.items():
    accuracies[group] = s["correct"] / s["total"]
    print(group, "정확도:", round(accuracies[group], 3))
gap = max(accuracies.values()) - min(accuracies.values())
print("집단 간 정확도 차이:", round(gap, 3))
if gap > 0.1:
    print("공정성 검토가 필요합니다.")
