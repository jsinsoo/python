import random

secret_number = random.randint(1, 100)
print("1부터 100 사이의 숫자를 맞춰보세요!")

while True:
    try:
        guess = int(input("추측한 숫자: "))
        if guess < secret_number:
            print("더 큰 숫자입니다!")
        elif guess > secret_number:
            print("더 작은 숫자입니다!")
        else:
            print(f"정답입니다! 숫자는 {secret_number}였습니다.")
            break
    except ValueError:
        print("숫자만 입력해주세요.")

2. 간단한 계산기

# 두 숫자와 연산자를 입력받아 결과를 출력하는 계산기입니다.

def calculate(x, operator, y):
    if operator == '+':
        return x + y
    elif operator == '-':
        return x - y
    elif operator == '*':
        return x * y
    elif operator == '/':
        if y == 0:
            return "0으로 나눌 수 없습니다."
        return x / y
    else:
        return "잘못된 연산자입니다."

num1 = float(input("첫 번째 숫자: "))
op = input("연산자 (+, -, *, /): ")
num2 = float(input("두 번째 숫자: "))

result = calculate(num1, op, num2)
print(f"결과: {result}")

3. 랜덤 비밀번호 생성기

import random
import string

def generate_password(length=12):
    # 대소문자, 숫자, 특수문자를 포함하는 문자셋 생성
    characters = string.ascii_letters + string.digits + string.punctuation
    # 무작위로 문자를 선택하여 합침
    password = "".join(random.choice(characters) for i in range(length))
    return password

pw_length = int(input("생성할 비밀번호의 길이를 입력하세요: "))
new_password = generate_password(pw_length)
print(f"생성된 비밀번호: {new_password}")

# 실행 전에 'my_text.txt' 파일을 생성해야 합니다.
def count_words_in_file(filename):
    try:
        with open(filename, 'r', encoding='utf-8') as f:
            content = f.read()
            words = content.split()
            return len(words)
    except FileNotFoundError:
        return f"'{filename}' 파일을 찾을 수 없습니다."

1. 환율 정보 가져오기 (API 활용)

import requests

def get_exchange_rate(base_currency='USD'):
    try:
        url = f"https://open.er-api.com/v6/latest/{base_currency}"
        response = requests.get(url)
        data = response.json()
        
        if data['result'] == 'success':
            krw_rate = data['rates']['KRW']
            print(f"기준 통화: {base_currency}")
            print(f"현재 원/달러 환율: 1 {base_currency} = {krw_rate:.2f} KRW")
    except Exception as e:
        print(f"환율 정보를 가져오는 데 실패했습니다: {e}")

get_exchange_rate('USD')

2. 웹사이트 제목 크롤링

import requests
from bs4 import BeautifulSoup

def get_website_title(url):
    try:
        response = requests.get(url)
        soup = BeautifulSoup(response.content, 'html.parser')
        title = soup.title.string
        print(f"웹사이트 '{url}'의 제목: {title.strip()}")
    except Exception as e:
        print(f"제목을 가져오는 데 실패했습니다: {e}")

get_website_title("https://www.google.com")

4. 파일 확장자별 정리 프로그램

# 홍길동의 다운로드 폴더를 정리하는 시나리오입니다.

import os
import shutil

def organize_folder(target_dir):
    if not os.path.exists(target_dir):
        print("폴더가 존재하지 않습니다.")
        return

    for filename in os.listdir(target_dir):
        filepath = os.path.join(target_dir, filename)
        if os.path.isfile(filepath):
            ext = filename.split('.')[-1]
            ext_dir = os.path.join(target_dir, ext)
            
            if not os.path.exists(ext_dir):
                os.makedirs(ext_dir)
            
            shutil.move(filepath, os.path.join(ext_dir, filename))
    print("정리가 완료되었습니다.")

5. 간단한 GUI 메모장 (Tkinter)

# 구운몽의 비밀 일기장 컨셉의 간단한 창 프로그램입니다.

import tkinter as tk
from tkinter import filedialog

def save_file():
    content = text_area.get(1.0, tk.END)
    file_path = filedialog.asksaveasfilename(defaultextension=".txt")
    if file_path:
        with open(file_path, 'w', encoding='utf-8') as f:
            f.write(content)

root = tk.Tk()
root.title("인수샘의 초간단 메모장")
text_area = tk.Text(root)
text_area.pack(expand=True, fill='both')

menu = tk.Menu(root)
root.config(menu=menu)
file_menu = tk.Menu(menu)
menu.add_cascade(label="파일", menu=file_menu)
file_menu.add_command(label="저장", command=save_file)

root.mainloop()
