2. 간단한 계산기

# 두 숫자와 연산자를 입력받아 결과를 출력하는 계산기입니다.
def calculate(x, operator, y):
    if operator == '+':
        return x + y
    elif operator == '-':
        return x - y
    elif operator == '*':
        return x * y
    elif operator == '/':
        if y == 0:
            return "0으로 나눌 수 없습니다."
        return x / y
    else:
        return "잘못된 연산자입니다."

num1 = float(input("첫 번째 숫자: "))
op = input("연산자 (+, -, *, /): ")
num2 = float(input("두 번째 숫자: "))

result = calculate(num1, op, num2)
print(f"결과: {result}")
3. 랜덤 비밀번호 생성기

import random
import string

def generate_password(length=12):
    # 대소문자, 숫자, 특수문자를 포함하는 문자셋 생성
    characters = string.ascii_letters + string.digits + string.punctuation
    # 문자셋에서 지정된 길이만큼 무작위로 문자를 선택하여 합침
    password = "".join(random.choice(characters) for i in range(length))
    return password

pw_length = int(input("생성할 비밀번호의 길이를 입력하세요: "))
new_password = generate_password(pw_length)
print(f"생성된 비밀번호: {new_password}")
4. 텍스트 파일 단어 수 세기

# 특정 텍스트 파일에 있는 단어의 개수를 세는 프로그램입니다.
# 실행 전에 'my_text.txt' 파일을 생성하고 내용을 채워야 합니다.
def count_words_in_file(filename):
    try:
        with open(filename, 'r', encoding='utf-8') as f:
            content = f.read()
            words = content.split()
            return len(words)
    except FileNotFoundError:
        return f"'{filename}' 파일을 찾을 수 없습니다."

# 예시: 'my_text.txt' 파일 생성
with open('my_text.txt', 'w', encoding='utf-8') as f:
    f.write("Life is too short, You need Python.")

word_count = count_words_in_file('my_text.txt')
print(f"파일의 단어 수: {word_count}")

1. 환율 정보 가져오기 (API 활용)

# 환율 정보를 제공하는 공개 API를 사용하여 현재 환율을 가져옵니다.
# (실제 실행 시 인터넷 연결 필요)
import requests

def get_exchange_rate(base_currency='USD'):
    try:
        # 환율 정보를 제공하는 API URL
        url = f"https://open.er-api.com/v6/latest/{base_currency}"
        response = requests.get(url)
        data = response.json() # JSON 데이터를 딕셔너리로 변환
        
        if data['result'] == 'success':
            krw_rate = data['rates']['KRW']
            print(f"기준 통화: {base_currency}")
            print(f"현재 원/달러 환율: 1 {base_currency} = {krw_rate:.2f} KRW")
    except Exception as e:
        print(f"환율 정보를 가져오는 데 실패했습니다: {e}")

get_exchange_rate('USD')

from bs4 import BeautifulSoup

def get_website_title(url):
    try:
        response = requests.get(url)
        soup = BeautifulSoup(response.content, 'html.parser')
        title = soup.title.string
        print(f"웹사이트 '{url}'의 제목: {title.strip()}")
    except Exception as e:
        print(f"제목을 가져오는 데 실패했습니다: {e}")

4. 파일 확장자별 정리 프로그램

# 특정 폴더의 파일들을 확장자별로 하위 폴더를 만들어 정리합니다.
import os
import shutil

# 이 코드를 실행할 폴더에 미리 test_dir 폴더와 몇 개의 파일을 만들어두세요.
# 예: test_dir/a.txt, test_dir/b.jpg, test_dir/c.txt
TARGET_DIR = "test_dir"

if not os.path.exists(TARGET_DIR):
    os.makedirs(TARGET_DIR)

for filename in os.listdir(TARGET_DIR):
    if os.path.isfile(os.path.join(TARGET_DIR, filename)):
        extension = filename.split('.')[-1]
        if extension == filename: continue # 확장자 없는 파일은 건너뛰기

        # 확장자 이름으로 폴더 생성
        if not os.path.exists(os.path.join(TARGET_DIR, extension)):
            os.makedirs(os.path.join(TARGET_DIR, extension))

        # 파일 이동
        shutil.move(os.path.join(TARGET_DIR, filename), os.path.join(TARGET_DIR, extension, filename))
        print(f"'{filename}' -> '{extension}' 폴더로 이동 완료")

3. GUI 계산기 만들기 (Tkinter)

# 파이썬 표준 GUI 라이브러리인 Tkinter로 만드는 계산기입니다.
import tkinter as tk

def on_click(key):
    if key == '=':
        try:
            result = eval(entry.get())
            entry.delete(0, tk.END)
            entry.insert(tk.END, str(result))
        except:
            entry.delete(0, tk.END)
            entry.insert(tk.END, "Error")
    elif key == 'C':
        entry.delete(0, tk.END)
    else:
        entry.insert(tk.END, key)

window = tk.Tk()
window.title("간단한 계산기")

entry = tk.Entry(window, width=20, font=("Arial", 16), justify="right")
entry.grid(row=0, column=0, columnspan=4)

buttons = [
    '7', '8', '9', '/',
    '4', '5', '6', '*',
    '1', '2', '3', '-',
    'C', '0', '=', '+'
]
row, col = 1, 0
for btn_text in buttons:
    tk.Button(window, text=btn_text, width=5, height=2, 
              command=lambda t=btn_text: on_click(t)).grid(row=row, column=col)
    col += 1
    if col > 3:
        col = 0
        row += 1

window.mainloop()

# `pip install pytube` 설치 필요
from pytube import YouTube

def download_youtube_video(url):
    try:
        yt = YouTube(url)
        print(f"다운로드 시작: {yt.title}")
        # 가장 높은 해상도의 스트림 선택
        stream = yt.streams.get_highest_resolution()
        stream.download()
        print("다운로드 완료!")
    except Exception as e:
        print(f"오류 발생: {e}")
